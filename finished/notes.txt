"use client";
import React, { useRef } from "react";
import { useState } from "react";

const InputOutput = () => {
  const [message, setMessage] = useState<string>("Hello World");
  const messageRef = useRef<string>("Hello World");

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    messageRef.current = e.target.value; // updates silently
  };
  return (
    <section className="flex flex-col w-full gap-5">
      <div className="flex flex-col gap-3 w-full h-[200px] bg-slate-900 rounded-xl p-3">
        <p>{message}</p>
        <p>{messageRef.current}</p>
      </div>
      <input
        type="text"
        className="w-full p-2 bg-slate-200 rounded-xl text-black focus:outline-none "
        onChange={handleChange}
      />
      <button
        className="p-5 bg-blue-500 text-white rounded-xl cursor-pointer"
        onClick={() => setMessage(messageRef.current)}
      >
        Send
      </button>
    </section>
  );
};
